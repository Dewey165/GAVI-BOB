CREATE TABLE movies (
movie_id SERIAL,
title VARCHAR(250),
year INT,
PRIMARY KEY (movie_id));

CREATE TABLE ratings (
id SERIAL,
movie_id INT references movies(movie_id),
user_id VARCHAR(250),
rating INT,
PRIMARY KEY (id));

CREATE TABLE genres (
id SERIAL,
name VARCHAR(250),
PRIMARY KEY (id));

CREATE TABLE moviegenres (
movie_id INT REFERENCES movies(movie_id),
genre_id INT REFERENCES genres(id),
PRIMARY KEY (movie_id, genre_id));
