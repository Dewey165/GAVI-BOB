CREATE TABLE movies (
movie_id char(9),
title VARCHAR(250),
year char(4),
PRIMARY KEY (movie_id));

CREATE TABLE ratings (
id char(9),
movie_id char(9) references movies(movie_id),
user_id VARCHAR(250),
rating char(2),
PRIMARY KEY (id));

CREATE TABLE genres (
id char(2),
name VARCHAR(250),
PRIMARY KEY (id));

CREATE TABLE moviegenres (
movie_id char(9) REFERENCES movies(movie_id),
genre_id char(2) REFERENCES genres(id),
PRIMARY KEY (movie_id, genre_id));
